Regex Validation

March 27, 2016 anmol-lf

Regular Expression aka Regex are a way to define patterns. It is particularly useful when we need to validate input to check if it matches a certain criteria or not. Example include email addresses, passwords, credit card numbers, etc.

let EmailRegex = "[A-Z0-9a-z._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,6}"
let CreditCardRegex = "[0-9]{16,20}"
 
func validateStringWithRegex(string: String, regex: String) -> Bool {
    guard (string.rangeOfString(regex, options: .RegularExpressionSearch) != nil) else {
        return false
    }
    return true
}
 
func validateEmail(email: String) -> Bool {
    return validateStringWithRegex(email, regex: EmailRegex)
}
 
func validateCreditCard(number: String) -> Bool {
    return validateStringWithRegex(number, regex: CreditCardRegex)
}
